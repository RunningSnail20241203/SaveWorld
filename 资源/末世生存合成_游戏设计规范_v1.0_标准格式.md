### 5.2 自动生成算法（伪代码）

```python
def generate_daily_orders(player):
    orders = []
    player_max_level = player.history.max_level
    
    for i in range(5):
        # 80%概率：生成 ≤ player_max_level + 2
        # 20%概率：生成 ≤ player_max_level
        
        target_level = random_weighted([
            (player_max_level, 0.2),
            (player_max_level + 1, 0.5),
            (player_max_level + 2, 0.3)
        ])
        
        item_type = random_item_by_level(target_level)
        quantity = get_quantity_by_level(target_level)
        reward = get_reward_by_level(target_level)
        
        orders.append({
            "item": item_type,
            "level": target_level,
            "quantity": quantity,
            "reward": reward
        })
    
    # 5%概率引导订单
    if random(100) < 5:
        new_item = get_random_l1_not_in_backpack(player)
        if new_item:
            orders.append(guide_order(new_item))
    
    return orders
```
